output.sh:

#! /bin/bash

FILE=$1
TOTALTWEETS=106569
ID=`awk 'FNR == 1{print FILENAME}' $FILE | cut -f1 -d "."`
for HASHTAG in `cat $FILE`;
do
    HOverall=`grep -w $HASHTAG sorted.hts | awk '{print $1}'`
    HinC=`sort $FILE | uniq -c | sort -nr -k1 | grep $HASHTAG | awk '{print $1}'`
    tweetCount=`wc -l $FILE | awk '{print $1}'`
    HEntire=`grep -w $HASHTAG sortedEntire.hts | awk '{print $1}'`
    freqHinC=`echo "scale=10; $HinC/$tweetCount" | bc`
    freqHOverall=`echo "scale=10; $HEntire/$TOTALTWEETS" | bc`
    relativeFreq=`echo "scale=10; $freqHinC/$freqHOverall" | bc`
    echo -e $HASHTAG "\t" $ID "\t" $relativeFreq "\t" $freqHinC "\t" $HinC "\t" $HOverall "\t" $HEntire "\t" $TOTALTWEETS
done

TOTALTWEETS was calculated manually for convenience. Otherwise, grep for each hashtag in both the overall and entire hashtag files, and then compute the occurence for the own file and then echo the results.

#! /bin/bash

for file in *.hashtags

done
